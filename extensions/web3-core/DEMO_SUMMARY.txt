🛠️ 常用命令
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
# 查看浏览器演示指南
cat extensions/web3-core/BROWSER_DEMO_GUIDE.md

# 检查服务器状态
lsof -i :8080 | grep LISTEN

# 停止服务器
lsof -ti :8080 | xargs kill

# 重启服务器
cd extensions/web3-core && python3 -m http.server 8080 &

# 在浏览器中重新打开
open http://localhost:8080/demo.html

# 运行 Node.js CLI 演示
node --import tsx demo.ts
